
**1) Create a virtual environment**

```bash
cd /Users/vuhung/Desktop/LangGraph-Adventures

# Ensure Python 3.12 is available
/opt/homebrew/bin/python3.12 --version

# Create the virtual environment in .venv
/opt/homebrew/bin/python3.12 -m venv .venv

# Activate it (zsh)
source .venv/bin/activate

# Confirm the venv is active
which python
python -V
```

**2) Install the packages**

```bash
# Upgrade packaging tools inside the venv
python -m pip install --upgrade pip setuptools wheel

# Install dependencies from requirements.txt (at project root)
pip install -r requirements.txt
```

**3) Verify**

```bash
# Confirm Python is from .venv and print version
which python
python -V

# Verify the installed packages and their versions
python - <<'PY'
import sys
from importlib import metadata

packages = [
    "langchain",
    "langchain-ollama",
    "pydantic",
    "rich",
]

print(f"Python: {sys.version}")
for name in packages:
    try:
        print(f"{name}: {metadata.version(name)}")
    except Exception as e:
        print(f"{name}: not found ({e})")
PY
```
